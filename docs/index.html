Quick Start

Get started with DDB in minutes

1. Install DDB

# Homebrew (Recommended)
brew tap watkinslabs/ddb
brew install ddb

# Go Install
go install github.com/watkinslabs/ddb@latest

# Enable shell completion
source <(ddb completion bash)

2. Query Files

# Basic query
ddb query "SELECT * FROM data WHERE amount > 1000" \
  --file data:./sales.csv

# Multi-table join
ddb query "SELECT u.name, o.total FROM users u JOIN orders o ON u.id = o.user_id" \
  --file users:./users.csv \
  --file orders:./orders.json

3. Export Results

# Export to Excel for business reports
ddb query "SELECT department, COUNT(*) as headcount, AVG(salary) as avg_pay FROM employees GROUP BY department" \
  --file employees:./employees.csv \
  --output excel \
  --export department_report.xlsx

# Export to Parquet for analytics
ddb query "SELECT * FROM transactions WHERE amount > 1000" \
  --file transactions:./data.csv.gz \
  --output parquet \
  --export high_value_transactions.parquet

# Query compressed files directly
ddb query "SELECT customer, SUM(amount) FROM sales GROUP BY customer" \
  --file sales:./sales.csv.gz \
  --output json \
  --export customer_totals.json
